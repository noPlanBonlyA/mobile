## 3. Dockerfiles

### 3.1. Main Backend Dockerfile

**Файл:** `docker/backend/Dockerfile`

```dockerfile
FROM python:3.12-slim as base

# Метаданные
LABEL maintainer="IT School Team"
LABEL description="Main Backend Service для IT School Platform"

# Установка системных зависимостей
RUN apt-get update && apt-get install -y \
    postgresql-client \
    curl \
    && rm -rf /var/lib/apt/lists/*

# Создание пользователя для приложения
RUN useradd -m -u 1000 appuser

WORKDIR /app

# Установка uv для управления Python зависимостями
RUN pip install --no-cache-dir uv

# Копирование файлов зависимостей
COPY backendITSchool/pyproject.toml backendITSchool/uv.lock ./

# Установка Python зависимостей
RUN uv sync --frozen

# Копирование кода приложения
COPY backendITSchool/ ./

# Изменение владельца файлов
RUN chown -R appuser:appuser /app

# Переключение на пользователя приложения
USER appuser

# Healthcheck
HEALTHCHECK --interval=30s --timeout=10s --start-period=40s --retries=3 \
    CMD curl -f http://localhost:8080/docs || exit 1

EXPOSE 8080

# Команда запуска
CMD ["uvicorn", "school_site.main:app", "--host", "0.0.0.0", "--port", "8080", "--workers", "4"]
```

**Особенности:**
- Использование Python 3.12 slim образа для минимизации размера
- Установка только необходимых системных пакетов
- Использование непривилегированного пользователя для безопасности
- Healthcheck для мониторинга состояния
- Многопроцессный запуск через workers

### 3.3. Entrypoint скрипты

**Файл:** `docker/entrypoint.sh`

```bash
#!/bin/bash
set -e

echo "Waiting for PostgreSQL to be ready..."
until pg_isready -h $DB_HOST -p $DB_PORT -U $DB_USER; do
    echo "PostgreSQL is unavailable - sleeping"
    sleep 2
done

echo "PostgreSQL is ready!"

echo "Running database migrations..."
alembic upgrade head

echo "Starting application..."
exec "$@"
```

**Делаем исполняемым:**
```bash
chmod +x docker/entrypoint.sh
```

## 4. Docker Compose

### 4.1. Полный docker-compose.yml

**Файл:** [docker-compose.yml](Production 46/docker-compose.yml)

```yaml
version: '3.8'

services:
  # PostgreSQL для Main Backend
  db:
    image: postgres:17-bookworm
    container_name: school_site_db
    environment:
      POSTGRES_DB: ${DB_DATABASE}
      POSTGRES_USER: ${DB_USER}
      POSTGRES_PASSWORD: ${DB_PASSWORD}
      PGDATA: /var/lib/postgresql/data/pgdata
    restart: unless-stopped
    ports:
      - "5435:5432"
    networks:
      - school_site-network
    volumes:
      - school_site_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${DB_USER}"]
      interval: 10s
      timeout: 5s
      retries: 5

  # PostgreSQL для Email Service
  email_db:
    image: postgres:17-bookworm
    container_name: email_db
    environment:
      POSTGRES_DB: ${DB_DATABASE}
      POSTGRES_USER: ${DB_USER}
      POSTGRES_PASSWORD: ${DB_PASSWORD}
      PGDATA: /var/lib/postgresql/data/pgdata
    restart: unless-stopped
    ports:
      - "5436:5432"
    networks:
      - school_site-network
    volumes:
      - email_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${DB_USER}"]
      interval: 10s
      timeout: 5s
      retries: 5

  # MinIO File Storage
  minio:
    image: minio/minio:latest
    container_name: minio_storage
    ports:
      - "9000:9000"    # API
      - "9001:9001"    # Web Console
    volumes:
      - minio_data:/data
    environment:
      MINIO_ROOT_USER: minioadmin
      MINIO_ROOT_PASSWORD: minioadmin
    command: server --console-address ":9001" /data
    networks:             
      - school_site-network
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9000/minio/health/live"]
      interval: 30s
      timeout: 20s
      retries: 3
    restart: unless-stopped

  # Main Backend Service
  main-app:
    build:
      context: .
      dockerfile: ./docker/backend/Dockerfile
      args:
        - BUILD_DATE=$(date -u +'%Y-%m-%dT%H:%M:%SZ')
        - VCS_REF=$(git rev-parse --short HEAD)
    container_name: school_site_backend
    restart: unless-stopped
    expose:
      - 8080
    ports:
      - "8080:8080"
    environment:
      DB_HOST: db
      DB_PORT: 5432
      DB_USER: ${DB_USER}
      DB_PASSWORD: ${DB_PASSWORD}
      DB_DATABASE: ${DB_DATABASE}
      EMAIL_SERVICE_URL: http://email-app:9090
      MINIO_URL: minio:9000
      MINIO_ACCESS_KEY: minioadmin
      MINIO_SECRET_KEY: minioadmin
    entrypoint: /docker/entrypoint.sh
    depends_on:
      db:
        condition: service_healthy
      minio:
        condition: service_healthy
    volumes: 
      - ./backendITSchool:/app
      - ./docker/entrypoint.sh:/docker/entrypoint.sh:ro
    networks:
      - school_site-network
    logging:
      driver: "json-file"
      options:
        max-size: "10m"
        max-file: "3"
  
  # Email Notification Service
  email-app:
    build:
      context: .
      dockerfile: ./docker/email/Dockerfile
    container_name: school_site_email_notifier
    restart: unless-stopped
    expose:
      - 9090
    ports:
      - "9090:9090"
    environment:
      DB_HOST: email_db
      DB_PORT: 5432
      DB_USER: ${DB_USER}
      DB_PASSWORD: ${DB_PASSWORD}
      DB_DATABASE: ${DB_DATABASE}
      SMTP_HOST: ${SMTP_HOST:-smtp.gmail.com}
      SMTP_PORT: ${SMTP_PORT:-587}
      SMTP_USER: ${SMTP_USER}
      SMTP_PASSWORD: ${SMTP_PASSWORD}
    entrypoint: /docker/entrypoint.sh
    depends_on:
      email_db:
        condition: service_healthy
    volumes: 
      - ./backendNotificationEmailService:/app
      - ./docker/entrypoint.sh:/docker/entrypoint.sh:ro
    networks:
      - school_site-network
    logging:
      driver: "json-file"
      options:
        max-size: "10m"
        max-file: "3"

networks:
  school_site-network:
    driver: bridge
    ipam:
      config:
        - subnet: 172.20.0.0/16

volumes:
  school_site_data:
    driver: local
  email_data:
    driver: local
  minio_data:
    driver: local
```

## 6. Сетевое взаимодействие

### 6.1. Docker Network Configuration

```yaml
networks:
  school_site-network:
    driver: bridge
    ipam:
      config:
        - subnet: 172.20.0.0/16
```

**Преимущества:**
- Изоляция от других Docker сетей
- Автоматический DNS между контейнерами
- Управляемая адресация

### 6.2. Service Discovery

Контейнеры обращаются друг к другу по именам сервисов:

```python
# Main Backend → Email Service
EMAIL_SERVICE_URL = "http://email-app:9090"

# Main Backend → MinIO
MINIO_CLIENT = Minio(
    "minio:9000",
    access_key="minioadmin",
    secret_key="minioadmin",
    secure=False
)

# Main Backend → PostgreSQL
DATABASE_URL = "postgresql://user:pass@db:5432/school_db"
```

### 6.3. Проброс портов

```yaml
ports:
  - "8080:8080"    # Main Backend API
  - "9090:9090"    # Email Service API
  - "9000:9000"    # MinIO API
  - "9001:9001"    # MinIO Console
  - "5435:5432"    # PostgreSQL Main (для внешнего доступа)
  - "5436:5432"    # PostgreSQL Email (для внешнего доступа)
```

## 7. Управление контейнерами

### 7.1. Основные команды

```bash
# Сборка образов
docker-compose build

# Сборка без кеша
docker-compose build --no-cache

# Запуск всех сервисов
docker-compose up -d

# Запуск конкретного сервиса
docker-compose up -d main-app

# Остановка всех сервисов
docker-compose down

# Остановка с удалением volumes
docker-compose down -v

# Просмотр логов
docker-compose logs -f

# Логи конкретного сервиса
docker-compose logs -f main-app

# Просмотр статуса
docker-compose ps

# Перезапуск сервиса
docker-compose restart main-app

# Масштабирование
docker-compose up -d --scale main-app=3
```

### 7.2. Работа с контейнерами

```bash
# Вход в контейнер
docker exec -it school_site_backend bash

# Выполнение команды в контейнере
docker exec school_site_backend alembic upgrade head

# Просмотр ресурсов
docker stats

# Инспекция контейнера
docker inspect school_site_backend

# Просмотр сетей
docker network ls
docker network inspect production_46_school_site-network
```

### 7.3. Управление образами

## 9. Production настройки

### 9.1. Оптимизация образов

**Multi-stage build пример:**

```dockerfile
# Стадия сборки
FROM python:3.12-slim as builder

WORKDIR /app
RUN pip install uv
COPY pyproject.toml uv.lock ./
RUN uv sync --frozen --no-dev

# Финальная стадия
FROM python:3.12-slim

WORKDIR /app
COPY --from=builder /app/.venv /app/.venv
COPY . .

ENV PATH="/app/.venv/bin:$PATH"

CMD ["uvicorn", "school_site.main:app", "--host", "0.0.0.0", "--port", "8080"]
```

### 9.2. Безопасность

```dockerfile
# Не использовать root
RUN useradd -m -u 1000 appuser
USER appuser

# Минимальный базовый образ
FROM python:3.12-slim

# Удаление кешей
RUN apt-get clean && rm -rf /var/lib/apt/lists/*

# Read-only файловая система
volumes:
  - ./app:/app:ro

# Ограничение ресурсов
deploy:
  resources:
    limits:
      cpus: '1'
      memory: 1G
    reservations:
      cpus: '0.5'
      memory: 512M
```
