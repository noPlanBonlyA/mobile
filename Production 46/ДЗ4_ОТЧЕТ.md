## 3. Базовые тесты

### 3.1. Тесты аутентификации

**Запрос: Successful Login**
```javascript
// Tests
pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response time is less than 1000ms", function () {
    pm.expect(pm.response.responseTime).to.be.below(1000);
});

pm.test("Response has user object", function () {
    var jsonData = pm.response.json();
    pm.expect(jsonData).to.be.an('object');
    pm.expect(jsonData).to.have.property('id');
    pm.expect(jsonData).to.have.property('username');
    pm.expect(jsonData).to.have.property('email');
    pm.expect(jsonData).to.have.property('role');
});

pm.test("Cookies are set", function () {
    pm.expect(pm.cookies.has('access_token')).to.be.true;
    pm.expect(pm.cookies.has('refresh_token')).to.be.true;
});

pm.test("User role is valid", function () {
    var jsonData = pm.response.json();
    pm.expect(['STUDENT', 'TEACHER', 'ADMIN']).to.include(jsonData.role);
});

// Сохранение токенов
var accessToken = pm.cookies.get('access_token');
var refreshToken = pm.cookies.get('refresh_token');
pm.environment.set("access_token", accessToken);
pm.environment.set("refresh_token", refreshToken);
pm.environment.set("user_id", pm.response.json().id);
```

**Запрос: Failed Login - Wrong Password**
```javascript
pm.test("Status code is 401", function () {
    pm.response.to.have.status(401);
});

pm.test("Error message is present", function () {
    var jsonData = pm.response.json();
    pm.expect(jsonData).to.have.property('detail');
    pm.expect(jsonData.detail).to.be.a('string');
});

pm.test("No tokens are set", function () {
    pm.expect(pm.cookies.has('access_token')).to.be.false;
});
```

**Запрос: Refresh Token**
```javascript
pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("New access token is provided", function () {
    var newToken = pm.cookies.get('access_token');
    pm.expect(newToken).to.not.be.empty;
    pm.environment.set("access_token", newToken);
});
```

### 3.2. CRUD тесты для пользователей

**Запрос: Create User**
```javascript
// Pre-request: генерация уникальных данных
var timestamp = Date.now();
pm.environment.set("test_username", `test_user_${timestamp}`);
pm.environment.set("test_email", `test${timestamp}@example.com`);

// Request body
{
  "username": "{{test_username}}",
  "email": "{{test_email}}",
  "password": "TestPassword123",
  "role": "STUDENT",
  "birth_date": "2000-01-01",
  "first_name": "Test",
  "surname": "User"
}

// Tests
pm.test("Status code is 201", function () {
    pm.response.to.have.status(201);
});

pm.test("User created with correct data", function () {
    var jsonData = pm.response.json();
    pm.expect(jsonData.username).to.eql(pm.environment.get("test_username"));
    pm.expect(jsonData.email).to.eql(pm.environment.get("test_email"));
    pm.expect(jsonData.role).to.eql("STUDENT");
});

pm.test("User has UUID", function () {
    var jsonData = pm.response.json();
    pm.expect(jsonData.id).to.match(/^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i);
    pm.environment.set("created_user_id", jsonData.id);
});

pm.test("Password is not in response", function () {
    var jsonData = pm.response.json();
    pm.expect(jsonData).to.not.have.property('password');
    pm.expect(jsonData).to.not.have.property('password_hash');
});
```

**Запрос: Get User by ID**
```javascript
pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("User data matches created user", function () {
    var jsonData = pm.response.json();
    pm.expect(jsonData.id).to.eql(pm.environment.get("created_user_id"));
    pm.expect(jsonData.username).to.eql(pm.environment.get("test_username"));
});

pm.test("Response has all required fields", function () {
    var jsonData = pm.response.json();
    var requiredFields = ['id', 'username', 'email', 'role', 'created_at', 'updated_at'];
    requiredFields.forEach(field => {
        pm.expect(jsonData).to.have.property(field);
    });
});
```

### 3.3. Тесты валидации

**Запрос: Create User - Invalid Email**
```javascript
// Request body
{
  "username": "test_user",
  "email": "invalid-email",  // Невалидный email
  "password": "TestPassword123",
  "role": "STUDENT",
  "birth_date": "2000-01-01"
}

// Tests
pm.test("Status code is 422", function () {
    pm.response.to.have.status(422);
});

pm.test("Validation error for email", function () {
    var jsonData = pm.response.json();
    pm.expect(jsonData.detail).to.be.an('array');
    var emailError = jsonData.detail.find(err => err.loc.includes('email'));
    pm.expect(emailError).to.exist;
});
```

```javascript
pm.test("Student progress lifecycle", function () {
    var jsonData = pm.response.json();
    
    // Студент записан на курс
    pm.expect(jsonData.courses.length).to.be.above(0);
    
    // Прогресс корректно рассчитывается
    var course = jsonData.courses[0];
    var expectedProgress = (course.completed_lessons / course.total_lessons) * 100;
    pm.expect(course.progress).to.be.closeTo(expectedProgress, 0.1);
    
    // История баллов содержит записи
    pm.expect(jsonData.points_history.length).to.be.above(0);
});
```

```javascript
pm.globals.set("generateUUID", () => {
    return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
        var r = Math.random() * 16 | 0, v = c == 'x' ? r : (r & 0x3 | 0x8);
        return v.toString(16);
    });
});
```

## 6. Newman - CLI Runner

### 6.1. Установка Newman

```bash
npm install -g newman
```

### 6.2. Запуск тестов через CLI

```bash
# Запуск коллекции
newman run IT_School_API_Tests.postman_collection.json \
  -e Local.postman_environment.json \
  -r cli,json,html

# С дополнительными опциями
newman run collection.json \
  --environment environment.json \
  --reporters cli,html \
  --reporter-html-export report.html \
  --delay-request 100 \
  --timeout-request 5000
```

### 6.3. Интеграция с CI/CD

```yaml
# .github/workflows/api-tests.yml
name: API Tests

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      
      - name: Start API Server
        run: |
          docker-compose up -d
          sleep 10
      
      - name: Install Newman
        run: npm install -g newman
      
      - name: Run API Tests
        run: |
          newman run tests/IT_School_API_Tests.postman_collection.json \
            -e tests/CI.postman_environment.json \
            --reporters cli,junit \
            --reporter-junit-export results.xml
      
      - name: Publish Test Results
        uses: EnricoMi/publish-unit-test-result-action@v1
        if: always()
        with:
          files: results.xml
```
