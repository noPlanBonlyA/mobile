## 2. Выбор технологии

### 2.1. RabbitMQ vs Kafka

**Выбрано: RabbitMQ**

**Причины:**
- ✅ Простота интеграции и настройки
- ✅ Поддержка различных паттернов обмена (fanout, topic, direct)
- ✅ Встроенный management UI
- ✅ Гарантии доставки сообщений
- ✅ Подходит для event-driven архитектуры

**Kafka** - лучше подходит для:
- High-throughput потоковой обработки
- Event sourcing
- Логирование и аналитика

### 5.2. RabbitMQ Connection Manager

**Файл:** `school_site/core/messaging/rabbitmq.py`

```python
import asyncio
from typing import Optional
from aio_pika import connect_robust, Connection, Channel, Exchange, ExchangeType
from aio_pika.pool import Pool
import logging

logger = logging.getLogger(__name__)

class RabbitMQManager:
    """Менеджер подключения к RabbitMQ"""
    
    def __init__(self, url: str):
        self.url = url
        self.connection_pool: Optional[Pool] = None
        self.channel_pool: Optional[Pool] = None
        self._exchange: Optional[Exchange] = None
    
    async def start(self):
        """Инициализация пулов подключений"""
        logger.info("Initializing RabbitMQ connection pools")
        
        # Пул подключений
        self.connection_pool = Pool(
            self._get_connection,
            max_size=10,
            loop=asyncio.get_event_loop()
        )
        
        # Пул каналов
        self.channel_pool = Pool(
            self._get_channel,
            max_size=50,
            loop=asyncio.get_event_loop()
        )
        
        # Создание exchange
        await self._setup_exchange()
        
        logger.info("RabbitMQ connection pools initialized")
    
    async def _get_connection(self) -> Connection:
        """Создание нового подключения"""
        return await connect_robust(self.url)
    
    async def _get_channel(self) -> Channel:
        """Получение канала из пула подключений"""
        async with self.connection_pool.acquire() as connection:
            return await connection.channel()
    
    async def _setup_exchange(self):
        """Создание exchange для событий"""
        async with self.channel_pool.acquire() as channel:
            self._exchange = await channel.declare_exchange(
                'school_events',
                ExchangeType.TOPIC,
                durable=True
            )
    
    async def publish(self, routing_key: str, message: dict):
        """
        Публикация сообщения
        
        Args:
            routing_key: ключ маршрутизации (например, 'user.created')
            message: данные сообщения (будут сериализованы в JSON)
        """
        import json
        from aio_pika import Message
        
        async with self.channel_pool.acquire() as channel:
            exchange = await channel.get_exchange('school_events')
            
            await exchange.publish(
                Message(
                    body=json.dumps(message).encode(),
                    content_type='application/json',
                    delivery_mode=2  # Persistent
                ),
                routing_key=routing_key
            )
            
            logger.info(f"Published message to {routing_key}: {message}")
    
    async def stop(self):
        """Закрытие всех подключений"""
        if self.channel_pool:
            await self.channel_pool.close()
        if self.connection_pool:
            await self.connection_pool.close()
        logger.info("RabbitMQ connection pools closed")
```

from uuid import UUID
from datetime import datetime

class EventPublisher:
    """Сервис публикации событий"""
    
    def __init__(self, rabbitmq_manager: RabbitMQManager):
        self.manager = rabbitmq_manager
    
    async def publish_user_created(self, user_id: UUID, email: str, username: str, role: str):

                    'email': email,
                    'username': username,
                    'role': role

        )

                    'student_id': str(student_id),

                    'course_name': course_name

## 6. Реализация Consumer (Email Service)

### 6.1. Message Handler

**Файл:** `notification_email_service/messaging/consumer.py`

```python
import asyncio
import json
from aio_pika import connect_robust, Message, IncomingMessage
from typing import Callable, Dict
import logging

logger = logging.getLogger(__name__)

class EventConsumer:
    """Consumer для обработки событий из RabbitMQ"""
    
    def __init__(self, rabbitmq_url: str):
        self.url = rabbitmq_url
        self.connection = None
        self.channel = None
        self.handlers: Dict[str, Callable] = {}
    
    def register_handler(self, event_type: str, handler: Callable):
        """Регистрация обработчика для типа события"""
        self.handlers[event_type] = handler
        logger.info(f"Registered handler for {event_type}")
    
    async def start(self):
        """Запуск consumer"""
        logger.info("Starting RabbitMQ consumer")
        
        # Подключение
        self.connection = await connect_robust(self.url)
        self.channel = await self.connection.channel()
        await self.channel.set_qos(prefetch_count=10)
        
        # Exchange
        exchange = await self.channel.declare_exchange(
            'school_events',
            type='topic',
            durable=True
        )
        
        # Queue для email уведомлений
        queue = await self.channel.declare_queue(
            'email_notifications',
            durable=True,
            arguments={
                'x-dead-letter-exchange': 'school_events_dlx',
                'x-dead-letter-routing-key': 'email.failed'
            }
        )
        
        # Binding для различных событий
        await queue.bind(exchange, routing_key='user.created')
        await queue.bind(exchange, routing_key='user.password_reset')
        await queue.bind(exchange, routing_key='course.enrolled')
        await queue.bind(exchange, routing_key='lesson.completed')
        await queue.bind(exchange, routing_key='points.awarded')
        
        # Начало потребления сообщений
        await queue.consume(self._process_message)
        
        logger.info("RabbitMQ consumer started")
    
    async def _process_message(self, message: IncomingMessage):
        """Обработка входящего сообщения"""
        async with message.process():
            try:
                # Парсинг сообщения
                body = json.loads(message.body.decode())
                event_type = body.get('event')
                
                logger.info(f"Received event: {event_type}")
                
                # Вызов соответствующего обработчика
                if event_type in self.handlers:
                    await self.handlers[event_type](body['data'])
                else:
                    logger.warning(f"No handler for event type: {event_type}")
                
            except Exception as e:
                logger.error(f"Error processing message: {e}", exc_info=True)
                # Сообщение будет отправлено в DLQ
                raise
    
    async def stop(self):
        """Остановка consumer"""
        if self.channel:
            await self.channel.close()
        if self.connection:
            await self.connection.close()
        logger.info("RabbitMQ consumer stopped")
```
