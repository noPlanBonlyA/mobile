## 3. Реализация CRUD-методов

### 3.1. Базовый репозиторий

**Файл:** [core/repositories/base_repository.py](Production 46/backendITSchool/school_site/core/repositories/base_repository.py)

```python
from typing import TypeVar, Generic, Type, List, Optional
from uuid import UUID
from sqlalchemy import select, update, delete, func
from sqlalchemy.ext.asyncio import AsyncSession

T = TypeVar('T')

class BaseRepository(Generic[T]):
    """Базовый репозиторий с CRUD операциями"""
    
    def __init__(self, model: Type[T], session: AsyncSession):
        self.model = model
        self.session = session
    
    async def create(self, **kwargs) -> T:
        """Создание записи"""
        instance = self.model(**kwargs)
        self.session.add(instance)
        await self.session.flush()
        await self.session.refresh(instance)
        return instance
    
    async def get_by_id(self, id: UUID) -> Optional[T]:
        """Получение записи по ID"""
        result = await self.session.execute(
            select(self.model).where(self.model.id == id)
        )
        return result.scalar_one_or_none()
    
    async def get_all(self, skip: int = 0, limit: int = 100) -> List[T]:
        """Получение всех записей с пагинацией"""
        result = await self.session.execute(
            select(self.model).offset(skip).limit(limit)
        )
        return result.scalars().all()
    
    async def update(self, id: UUID, **kwargs) -> Optional[T]:
        """Обновление записи"""
        await self.session.execute(
            update(self.model).where(self.model.id == id).values(**kwargs)
        )
        await self.session.flush()
        return await self.get_by_id(id)
    
    async def delete(self, id: UUID) -> bool:
        """Удаление записи"""
        result = await self.session.execute(
            delete(self.model).where(self.model.id == id)
        )
        await self.session.flush()
        return result.rowcount > 0
    
    async def count(self) -> int:
        """Подсчет количества записей"""
        result = await self.session.execute(
            select(func.count()).select_from(self.model)
        )
        return result.scalar_one()
    
    async def exists(self, id: UUID) -> bool:
        """Проверка существования записи"""
        result = await self.session.execute(
            select(func.count()).select_from(self.model).where(self.model.id == id)
        )
        return result.scalar_one() > 0
```

```python
from typing import Optional, List
from sqlalchemy import select
from sqlalchemy.orm import selectinload
from sqlalchemy.ext.asyncio import AsyncSession
from school_site.core.repositories.base_repository import BaseRepository
from school_site.apps.students.models import Student
from uuid import UUID

class StudentRepository(BaseRepository[Student]):
    def __init__(self, session: AsyncSession):
        super().__init__(Student, session)
    
    async def get_by_user_id(self, user_id: UUID) -> Optional[Student]:
        """Получение студента по user_id"""
        result = await self.session.execute(
            select(Student).where(Student.user_id == user_id)
        )
        return result.scalar_one_or_none()
    
    async def get_with_groups(self, student_id: UUID) -> Optional[Student]:
        """Получение студента с загруженными группами"""
        result = await self.session.execute(
            select(Student)
            .options(selectinload(Student.groups))
            .where(Student.id == student_id)
        )
        return result.scalar_one_or_none()
    
    async def get_with_courses(self, student_id: UUID) -> Optional[Student]:
        """Получение студента с загруженными курсами"""
        result = await self.session.execute(
            select(Student)
            .options(selectinload(Student.courses))
            .where(Student.id == student_id)
        )
        return result.scalar_one_or_none()
    
    async def add_points(self, student_id: UUID, points: int) -> Optional[Student]:
        """Добавление баллов студенту"""
        student = await self.get_by_id(student_id)
        if student:
            student.points += points
            await self.session.flush()
            await self.session.refresh(student)
        return student
    
    async def subtract_points(self, student_id: UUID, points: int) -> Optional[Student]:
        """Вычитание баллов студента"""
        student = await self.get_by_id(student_id)
        if student and student.points >= points:
            student.points -= points
            await self.session.flush()
            await self.session.refresh(student)
        return student
```
